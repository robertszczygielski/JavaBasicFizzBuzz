						public abstract class FizzBuzzGame implements Game {

							abstract protected String getFor3();
							abstract protected String getFor5();
					

						return "Fuzz" -> return getFor3();
						return "Buzz" -> return getFor5();
					

						if (it % 15 == 0) return getFor3()+getFor5();
						if (it % 3 == 0) return getFor3();
						if (it % 5 == 0) return getFor5();
